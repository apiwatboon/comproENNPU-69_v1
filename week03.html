ทำไมต้องภาษาซี?

แล้ว float เก็บทศนิยมยังไง?

float (32 บิต) ไม่ได้เก็บเป็นเลขตรง ๆ แต่แยกเป็น 3 ส่วนแบบ “วิทยาศาสตร์” (มาตรฐาน IEEE 754): 1 บิตเครื่องหมาย + 8 บิตเลขชี้กำลัง (exponent) + 23 บิตค่าตัวเลข (mantissa) — จึงเก็บได้ทั้งเล็กมากและใหญ่มาก แลกกับการ “ปัดเศษ” (double = 64 บิต ละเอียดกว่า)

char ก็คือจำนวนเต็ม 8 บิต

char เก็บ 1 ไบต์ = ตัวเลข 0–255 (unsigned) หรือ −128–127 (signed) — เวลาเก็บ 'A' จริง ๆ คือเก็บเลข 65 ตามรหัส ASCII (คาบที่ 2!) นั่นเอง ตัวอักษรกับจำนวนเต็มเล็ก ๆ จึงเป็นสิ่งเดียวกันในภาษาซี

ตัวอย่างที่ 4.2 — about_me.c
#include <stdio.h>

int main() {
    int age = 19;          // จำนวนเต็ม
    float gpa = 3.45;      // ทศนิยม
    char grade = 'B';      // ตัวอักษรเดี่ยวใน ' '
    age = age + 1;         // วันเกิด! ค่าใหม่ทับค่าเดิม
    printf("อายุ %d ปี เกรด %.2f ระดับ %c\n", age, gpa, grade);
    return 0;
}

สังเกตบรรทัด age = age + 1; — เครื่องหมาย = คือ "กำหนดค่าให้" ไม่ใช่เท่ากับทางคณิตศาสตร์: คำนวณฝั่งขวาก่อน (19 + 1 = 20) แล้วเก็บกลับเข้ากล่อง

Arrays

อาร์เรย์ (Array) — ตัวแปรเดียวที่มีหลายกล่อง

แทนที่จะประกาศตัวแปรทีละตัว (a, b, c, …) — อาร์เรย์ คือตัวแปรเดียวที่จอง “หลายกล่องเรียงติดกัน” เข้าถึงแต่ละกล่องด้วย index (เริ่มที่ 0) · อาร์เรย์มี “หลายชั้น (มิติ)” ได้ — ด้านล่างมี 3 ชุดให้เทียบกัน แต่ละชุดปรับขนาดเองได้ ชี้/กดกล่องเพื่อดู index (สูงสุด 10 ต่อแกน · 3 มิติ 5 ต่อแกน)

1 มิติ อาร์เรย์แถวเดียว — box[คอลัมน์]
ชี้หรือกดที่กล่อง เพื่อดู index
2 มิติ เพิ่มแถว (row × col) — box[แถว][คอลัมน์]
ชี้หรือกดที่กล่อง เพื่อดู index
3 มิติ เพิ่มชั้น z (layer × row × col) — box[ชั้น][แถว][คอลัมน์]
ชี้หรือกดที่กล่อง เพื่อดู index

มาตรฐาน C (row-major): วงเล็บแรก = นอกสุด (ก้อนใหญ่: ชั้น) · วงเล็บสุดท้าย = ช่องเดี่ยวที่เรียงติดกันในหน่วยความจำ (คอลัมน์ ซ้าย→ขวา)

อ่านวงเล็บทีละชั้น — box[ชั้น][แถว][คอลัมน์] เลือกอะไรบ้าง?

ตั้งดัชนีที่อยากดู แล้วกด “ถัดไป” เพื่อดูการไล่เลือกทีละวงเล็บ — จากก้อนใหญ่ (ชั้นทั้งแผ่น) → แคบลงเป็นแถว → เหลือช่องเดียว

วิธีประกาศ และ กำหนดค่าอาร์เรย์

ประกาศ: ชนิด ชื่อ[ขนาด]; — เช่น int box[5]; จอง 5 กล่อง เข้าถึงด้วย index 0 ถึง 4 · กำหนดค่าได้ 2 แบบ: ใส่ตอนประกาศด้วย { } หรือกำหนดทีละกล่องด้วย box[i] = ค่า;

กฎที่ต้องจำ

index เริ่มที่ 0 — box[5] มีกล่อง box[0], box[1], …, box[4] (กล่องสุดท้ายคือ ขนาด − 1) · ห้ามอ้างเกินขอบเขต เช่น box[5] ในอาร์เรย์ 5 ช่อง (ภาษาซีไม่เตือน แต่ได้ค่าขยะ/พัง) · ขนาด ตายตัว ประกาศแล้วเปลี่ยนไม่ได้

กำหนดค่า 2 แบบ

1) ตอนประกาศ: int box[5] = {10, 20, 30, 40, 50};
2) ทีละกล่องภายหลัง: box[2] = 99; (แก้เฉพาะกล่อง index 2)
อ่านค่า: printf("%d", box[2]);

ตัวอย่าง — ประกาศพร้อมกำหนดค่าเริ่มต้นด้วย { }
#include <stdio.h>

int main() {
    int box[5] = {10, 20, 30, 40, 50};   // จอง 5 กล่อง ใส่ค่าเลย
    printf("box[0] = %d\n", box[0]);   // กล่องแรก
    printf("box[4] = %d\n", box[4]);   // กล่องสุดท้าย
    return 0;
}
ตัวอย่าง — ประกาศก่อน แล้วกำหนดค่าทีละกล่อง (box[i] = …)
#include <stdio.h>

int main() {
    int score[3];        // จอง 3 กล่อง (ยังเป็นค่าขยะ)
    score[0] = 85;       // กำหนดค่าทีละ index
    score[1] = 90;
    score[2] = 78;
    printf("score[1] = %d\n", score[1]);
    printf("รวม = %d\n", score[0] + score[1] + score[2]);
    return 0;
}

ตัวอย่างโปรแกรม — กดรันได้จริงในเบราว์เซอร์

รวมทุกอย่างในคาบนี้: ประกาศตัวแปร + ตัวดำเนินการ → โปรแกรมที่รันได้จริง กดปุ่มรัน (หรือกด แก้โค้ดแล้วรันซ้ำ) — คอมไพล์จริงด้วย GCC ออนไลน์

ตัวอย่างที่ 3.1 — rectangle.c (ตัวแปร int + การคูณ)
#include <stdio.h>

int main() {
    int w = 8, h = 5;
    int area = w * h;         // ตัวดำเนินการคูณ
    printf("พื้นที่ = %d ตร.หน่วย\n", area);
    return 0;
}

int main() {
    int a = 7, b = 2;
    printf("7 / 2   (int)   = %d\n", a / b);   // เศษหาย!
    printf("7 %% 2   (เศษ)   = %d\n", a % b);
    printf("7.0 / 2 (float) = %.1f\n", 7.0 / b);
    return 0;
}

#include <stdio.h>

int main() {

    printf("รวมสินค้า  = %d บาท\n", subtotal);

    return 0;
}
