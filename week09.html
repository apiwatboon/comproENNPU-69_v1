สัปดาห์ที่ 7 — เปิดครึ่งหลังของคอร์ส

บทที่ 9
ฟังก์ชัน

ครึ่งแรกคุณเขียนโปรแกรมเป็นแล้ว — ครึ่งหลังจะเขียนให้ "เป็นระเบียบ" ขึ้น เริ่มจากเครื่องมือจัดระเบียบที่สำคัญที่สุด: ฟังก์ชัน คือการห่อโค้ดเป็นกล่อง ตั้งชื่อ ระบุชนิดค่าที่คืน แล้วเรียกใช้ซ้ำกี่ครั้งก็ได้ — เขียนครั้งเดียว ใช้ทั้งโปรเจกต์

⏱️ ใช้เวลาเรียน ~2.5 ชั่วโมง 🏭 เครื่องจักรฟังก์ชัน + ตัวไล่เส้นทางการเรียก 📝 แบบทดสอบ 10 ข้อ

Learning Objectives

วัตถุประสงค์การเรียนรู้

นิยามฟังก์ชันภาษาซีพร้อมระบุชนิดค่าที่คืน (int, float, void) ได้ถูกต้อง
ส่งข้อมูลเข้าฟังก์ชันผ่านพารามิเตอร์ และรับผลลัพธ์กลับด้วย return ได้
อธิบายความแตกต่างระหว่าง return กับ printf ได้อย่างถูกต้อง
อธิบายขอบเขตตัวแปร (local / global) และผลของมันต่อการเขียนโปรแกรมได้
แยกโปรแกรมที่เคยเขียนในบทก่อน ๆ ออกเป็นฟังก์ชันย่อยที่ใช้ซ้ำได้
Defining Functions

📦 นิยามฟังก์ชัน — ห่อโค้ดใส่กล่อง ตั้งชื่อ แล้วเรียกใช้

โครงสร้าง: ชนิดค่าที่คืน ชื่อฟังก์ชัน(พารามิเตอร์) { ... } — ถ้าไม่คืนค่าใช้ void — กฎเหล็กข้อแรก: โค้ดในฟังก์ชันยังไม่ทำงาน จนกว่าจะถูก "เรียก" ด้วยชื่อ + ( )

ตัวอย่างที่ 9.1 — welcome.c (ฟังก์ชัน void: ทำงานแต่ไม่คืนค่า)
#include <stdio.h>

void show_header() {        // void = ไม่คืนค่า แค่ทำงาน
    printf("================================\n");
    printf(" ระบบคลังสินค้า v1.0\n");
    printf("================================\n");
}

int main() {
    show_header();          // เรียกครั้งที่ 1
    printf("...ทำงานบางอย่าง...\n");
    show_header();          // เรียกซ้ำ ไม่ต้องเขียน 3 บรรทัดนั้นใหม่
    return 0;
}
♻️

เขียนครั้งเดียว ใช้ร้อยครั้ง

โค้ด 3 บรรทัดถูกเรียกใช้ 2 ครั้งโดยพิมพ์แค่ชื่อ — ถ้าวันหนึ่งอยากเปลี่ยนหัวรายงาน ก็แก้ "ที่เดียว" มีผลทุกจุดที่เรียก

🏷️

ต้องบอกชนิดค่าที่คืนเสมอ

คืนจำนวนเต็ม → int · คืนทศนิยม → float · ไม่คืนอะไรเลย → void — main เองก็คือฟังก์ชันที่คืน int (return 0 = จบเรียบร้อย)

📛

นิยามไว้ "ก่อน" main

คอมไพเลอร์อ่านจากบนลงล่าง — ฟังก์ชันต้องถูกนิยาม (หรือประกาศโครง) ก่อนจุดที่เรียกใช้ ธรรมเนียมเริ่มต้น: เขียนฟังก์ชันไว้เหนือ main และตั้งชื่อเป็นคำกริยา เช่น calc_bmi, show_report

ตัวอย่างที่ 9.2 — add.c (ฟังก์ชันคืนค่าตัวแรกในชีวิต)
#include <stdio.h>

int add(int a, int b) {     // รับ int สองตัว คืน int หนึ่งตัว
    return a + b;
}

int main() {
    printf("%d\n", add(2, 3));      // ใช้ผลลัพธ์ตรง ๆ
    int x = add(10, 30);            // หรือเก็บไว้ใช้ต่อ
    printf("%d\n", x + 2);
    return 0;
}
Call Flow

🧵 เส้นทางการทำงาน — โปรแกรม "กระโดด" อย่างไรเมื่อเรียกฟังก์ชัน

นี่คือจุดที่มือใหม่งงที่สุด: โปรแกรมไม่ได้ไหลบนลงล่างเฉย ๆ อีกแล้ว กด "รันทีละขั้น ▶" ดูการกระโดด ขึ้น–ลง ระหว่าง main กับตัวฟังก์ชัน

3int main() {

Parameters & Return

🏭 เครื่องจักรฟังก์ชัน: ใส่วัตถุดิบ → ได้ผลผลิต

ฟังก์ชันที่ดีเหมือนเครื่องจักร: รับวัตถุดิบทาง พารามิเตอร์ แปรรูปข้างใน แล้วส่งผลผลิตกลับด้วย return — เลือกเครื่องจักร ใส่ค่า แล้วกดเดินเครื่อง

อาร์กิวเมนต์ (ขาเข้า)25
→
c_to_f(c)กำลังรอวัตถุดิบ
→
ค่าที่ return (ขาออก)?
นิยามของเครื่องจักรตัวนี้
float c_to_f(float c) {
    return c * 9 / 5 + 32;
}

🆚 return ไม่ใช่ printf!

return = ส่งค่ากลับให้ "โค้ดที่เรียก" เอาไปใช้ต่อได้ (เก็บใส่ตัวแปร, คำนวณต่อ)
printf = แค่โชว์ขึ้นจอให้ "คน" ดู โปรแกรมเอาไปใช้ต่อไม่ได้

float f = c_to_f(25); ✅ f เก็บ 77.0 ไว้คำนวณต่อ
ฟังก์ชันที่ printf แต่ "ไม่ return" ควรประกาศเป็น void — เอาผลไปเก็บใส่ตัวแปรไม่ได้!

📚 ศัพท์ที่ข้อสอบชอบถาม

พารามิเตอร์ (Parameter) = ตัวแปรรับค่าใน "นิยาม": float c_to_f(float c)
อาร์กิวเมนต์ (Argument) = ค่าจริงตอน "เรียก": c_to_f(25)

จำง่าย ๆ: พารามิเตอร์คือ "ช่องรับ" อาร์กิวเมนต์คือ "ของที่หย่อนลงช่อง" — ส่งได้หลายตัวคั่นด้วยจุลภาค ตามลำดับซ้ายไปขวา และทุกตัวต้องระบุชนิด

ตัวอย่างที่ 9.3 — grade_fn.c (ห่อตรรกะบทที่ 6 เป็นฟังก์ชัน)
char grade(int score) {
    if (score >= 80) return 'A';
    else if (score >= 70) return 'B';
    else if (score >= 60) return 'C';
    else if (score >= 50) return 'D';
    else return 'F';
}

int main() {
    printf("%c %c %c\n", grade(85), grade(72), grade(40));
    return 0;
}
ตัวอย่างที่ 9.4 — max3.c (หาค่ามากสุดของ 3 จำนวน)
int max3(int a, int b, int c) {
    int m = a;              // ให้ตัวแรกเป็นแชมป์ก่อน
    if (b > m) m = b;
    if (c > m) m = c;
    return m;
}

int main() {
    printf("%d\n", max3(7, 9, 3));
    printf("%d\n", max3(42, 10, 17));
    return 0;
}
Variable Scope

🚪 ขอบเขตตัวแปร — ของในห้องใคร ห้องมัน

ตัวแปรที่ประกาศ "ในฟังก์ชัน" เป็นของฟังก์ชันนั้น (local) — หายไปทันทีที่ฟังก์ชันจบ โลกภายนอกมองไม่เห็น นี่ไม่ใช่ข้อจำกัด แต่คือระบบป้องกันตัวแปรตีกันในโปรแกรมใหญ่

ตัวอย่างที่ 9.5 — scope.c (ตัวอย่างที่ "ตั้งใจให้พัง")
float calc_vat(float price) {
    float vat = price * 0.07;   // vat เป็น local
    return price + vat;
}

int main() {
    float total = calc_vat(100);
    printf("%.2f\n", total);
    printf("%.2f\n", vat);   // บรรทัดนี้พัง!
    return 0;
}

🔑 สามกฎของขอบเขต

1️⃣ ตัวแปรที่ประกาศในฟังก์ชัน (รวมพารามิเตอร์) ใช้ได้เฉพาะในฟังก์ชันนั้น
2️⃣ อยากได้ค่าออกมา → ส่งผ่าน return เท่านั้น อย่าหวังพึ่งตัวแปรข้างใน
3️⃣ ตัวแปร global (ประกาศนอกทุกฟังก์ชัน) มีจริงแต่ควรเลี่ยง — รับทุกอย่างผ่านพารามิเตอร์ ชีวิตจะไม่งง

💡 เรียกซ้ำ = เกิดใหม่: ตัวแปร local ถูกสร้างใหม่ทุกครั้งที่ฟังก์ชันถูกเรียก ไม่จำค่าจากรอบก่อน

🛠️

ย้อนปรับปรุงของเก่า: bmi.c ฉบับมีฟังก์ชัน

โปรแกรม BMI บทที่ 4 เขียนใหม่ให้เป็นระเบียบ: float calc_bmi(float w, float h) { return w / (h * h); } แล้ว main เหลือแค่ รับค่า → เรียกฟังก์ชัน → แสดงผล — โปรแกรมโลจิสติกส์คำนวณค่าส่ง 50 จุดทั่วประเทศ ก็คือฟังก์ชันค่าส่งหนึ่งตัวถูกเรียก 50 ครั้งนั่นเอง

Check Your Understanding

📝 แบบทดสอบท้ายบท (10 ข้อ)

แยกให้ออก: นิยาม vs เรียกใช้ · พารามิเตอร์ vs อาร์กิวเมนต์ · return vs printf · void คืออะไร

📘 อ่านก่อนสอบ · สรุปประเด็นสำคัญ

เข้าใจ "การนิยาม–เรียกใช้ฟังก์ชัน", ความต่าง return vs printf, และ scope ของตัวแปร แล้วฝึก trace ค่าที่คืนกลับ

1

รูปแบบการนิยามฟังก์ชัน

ชนิดที่คืนค่า ชื่อ(พารามิเตอร์) { ... }

float shipping_fee(float weight) {
    return weight * 20;
}

2

void = ไม่คืนค่า

void show_menu() หมายถึงฟังก์ชันนี้ทำงาน (เช่นพิมพ์เมนู) แต่ไม่ส่งค่ากลับ

3

return ต่างจาก printf

return = ส่งค่ากลับให้ผู้เรียกเอาไปใช้ต่อ/คำนวณต่อ · printf = แค่แสดงผลบนจอ ไม่ได้ส่งค่าให้ใคร

4

การเรียกและไล่ค่า

แทนค่าอาร์กิวเมนต์เข้าพารามิเตอร์ แล้วเอาค่าที่ return ไปคำนวณต่อ

int twice(int x){ return x*2; }
printf("%d", twice(3) + 1);  // 6+1 = 7

int x = add(2,3);   // 5
int y = add(x,4);   // 9

5

ตัวแปรเฉพาะที่ (local) + ทำไมนิยามเหนือ main

ตัวแปรที่ประกาศในฟังก์ชันใช้ได้เฉพาะในฟังก์ชันนั้น · นิยามฟังก์ชันไว้เหนือ main เพื่อให้คอมไพเลอร์ "รู้จัก" ก่อนถูกเรียก

💡

จุดที่ข้อสอบชอบถาม

โจทย์ไล่ค่าที่ return ซ้อนกัน เช่น add(add(2,3),4) และเรื่อง scope — ตัวแปรในฟังก์ชันหนึ่งใช้ข้ามไปอีกฟังก์ชันไม่ได้

สิ่งที่ได้เรียน

นิยาม: ชนิดค่าที่คืน ชื่อ(พารามิเตอร์) { } — โค้ดข้างในรอจนถูกเรียก · void = ไม่คืนค่า · พารามิเตอร์รับของเข้า อาร์กิวเมนต์คือของจริงที่ส่ง · return ส่งค่ากลับไปใช้ต่อ (ต่างจาก printf ที่แค่โชว์) · ตัวแปร local เกิดและตายในฟังก์ชัน · นิยามฟังก์ชันไว้ก่อน main
