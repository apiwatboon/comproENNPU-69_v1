Learning Objectives

เขียนคำสั่ง if พร้อมเงื่อนไขใน ( ) และบล็อกใน { } ได้ถูกต้อง
เขียนโปรแกรมสองทางเลือกด้วย if–else ได้
เขียนโปรแกรมหลายทางเลือกด้วย if–else if–else และอธิบายว่าทำไมลำดับเงื่อนไขจึงสำคัญ
ใช้ switch–case กับทางเลือกแบบค่าตายตัว และอธิบายหน้าที่ของ break ได้
เขียนเงื่อนไขซ้อน (nested if) และประยุกต์กับโจทย์จริง เช่น การตัดเกรดและค่าไฟฟ้าแบบขั้นบันได
Program Structure

โครงสร้างโปรแกรมภาษาซี — บ้านที่ต้องมีครบทุกเสา

โปรแกรมซีทุกโปรแกรมในโลกมีโครงเดียวกัน: #include เรียกคลังคำสั่ง → int main() จุดเริ่มทำงาน → คำสั่งใน { } จบทุกคำสั่งด้วย ; — ลองรันโปรแกรมแรกในชีวิตเลย!

ตัวอย่างที่ 4.1 — hello.c
#include <stdio.h>      // เรียกคลังคำสั่งมาตรฐาน (มี printf อยู่ในนี้)

int main() {              // จุดเริ่มต้น — ทุกโปรแกรมเริ่มที่ main เสมอ
    printf("Hello, World!\n");
    printf("สวัสดีชาวโลก จากภาษาซี\n");
    return 0;             // บอกระบบว่าจบอย่างเรียบร้อย
}

เซมิโคลอน ; คือจุดจบประโยค

ทุกคำสั่งต้องปิดด้วย ; — ลืมตัวเดียว คอมไพเลอร์ฟ้อง error ทั้งโปรแกรมไม่ยอมรันเลย เป็นบั๊กอันดับหนึ่งของมือใหม่ภาษาซีทุกยุคทุกสมัย

ตัวเล็ก-ใหญ่ ไม่เหมือนกัน

ต้องพิมพ์ printf และ main ตัวเล็กเท่านั้น — พิมพ์ Printf หรือ Main คอมไพเลอร์ไม่รู้จักทันที

คอมเมนต์ // และ /* */

ข้อความหลัง // คือโน้ตสำหรับมนุษย์ คอมไพเลอร์ข้ามให้ (หลายบรรทัดใช้ /* ... */) — เขียนอธิบายโค้ดให้เป็นนิสัยตั้งแต่วันแรก

The if Statement

🔶 if — ข้าวหลามตัดฉบับโค้ดจริง

โครงสร้าง: if (เงื่อนไข) { คำสั่ง } — เงื่อนไขอยู่ใน ( ) เสมอ ถ้าค่าเป็นจริง (ไม่ใช่ 0) ทำบล็อกใน { } ถ้าเป็น 0 ข้ามไปเลย

ตัวอย่างที่ 6.1 — speed.c (สมมุติผู้ใช้พิมพ์ 120)
int speed;
scanf("%d", &speed);
if (speed > 90) {
    printf("เกินกำหนด! โดนใบสั่ง\n");   // อยู่ใน { } = อยู่ใน if
    printf("ค่าปรับ 1,000 บาท\n");
}
printf("ขับขี่ปลอดภัยนะครับ\n");        // นอก { } = ทำเสมอ
🧱

{ } คือขอบเขตของเงื่อนไข

ภาษาซีใช้วงเล็บปีกกาบอกว่าบรรทัดไหนอยู่ใน if — ถ้ามีคำสั่งเดียวละ { } ได้ แต่มือใหม่ ควรใส่เสมอ กันบั๊กตอนเพิ่มบรรทัดทีหลัง

☠️

อย่าใส่ ; ท้าย if!

if (speed > 90); — เซมิโคลอนตัวนั้นคือ "คำสั่งว่าง" ทำให้เงื่อนไข จบทันที และบล็อกข้างล่างทำงานเสมอ! คอมไพล์ผ่านด้วย — เป็นบั๊กเงียบสุดคลาสสิกของภาษาซี

🔗

ต่อจากคาบที่แล้วเป๊ะ ๆ

ในวงเล็บใส่อะไรก็ได้ที่ให้ค่า 1/0: speed > 90, n % 2 == 0, a > 0 && b > 0 — ทุกตัวจากคาบที่ 5 ใช้ได้หมด

Two-way Selection

⚖️ if–else — สองทาง เลือกทางเดียวเสมอ

เมื่อต้องทำ "อย่างใดอย่างหนึ่ง" เสมอ ใช้ else รับกรณีที่เงื่อนไขเป็นเท็จ — เหมือนผังงาน BMI ที่แตกสองทางจากข้าวหลามตัดแล้วไหลมารวมกันก่อนจบ

ตัวอย่างที่ 6.2 — even_odd.c (สมมุติผู้ใช้พิมพ์ 7)
int n;
scanf("%d", &n);
if (n % 2 == 0) {
    printf("%d เป็นเลขคู่\n", n);
} else {
    printf("%d เป็นเลขคี่\n", n);
}

🧠 สามกฎทองของ else

1️⃣ else ไม่มีเงื่อนไขของตัวเอง — มันคือ "กรณีที่เหลือทั้งหมด" (เขียน else (x < 50) ไม่ได้)
2️⃣ else ต้องตามหลังบล็อก } ของ if คู่ของมันทันที
3️⃣ สองบล็อกนี้ทำได้ "ทางเดียวเท่านั้น" ไม่มีทางทำทั้งคู่ — เหมือนตัวจำลองผังงานคาบที่ 3 ที่เส้นทางจริง/เท็จ ไม่เคยถูกเดินพร้อมกัน

Multi-way Selection

🎓 if–else if–else — เลือกหลายทาง: เครื่องตัดเกรด

เมื่อทางเลือกมีมากกว่าสอง ใช้ else if ต่อกี่ชั้นก็ได้ — คอมไพเลอร์ไล่ตรวจ จากบนลงล่าง เจอจริงตัวแรกทำแล้วออกจากโครงสร้างทันที ใส่คะแนนแล้วกด "ตรวจเกรด ▶" ดูการไล่เงื่อนไขแบบสด ๆ

🎛️ แผงควบคุม

if (score >= 80)      // เกรด A
else if (score >= 70) // เกรด B
else if (score >= 60) // เกรด C
else if (score >= 50) // เกรด D
else                     // เกรด F

ผลการตัดเกรด

?
ใส่คะแนนแล้วกดตรวจ — สังเกตว่าเงื่อนไขที่เป็นเท็จจะจางลง และโปรแกรมหยุดที่เงื่อนไขจริงตัวแรก
🚨

ลำดับเงื่อนไขสำคัญมาก!

ลองสลับเอา if (score >= 50) ขึ้นบรรทัดแรกดูสิ — คนได้ 95 คะแนนจะโดนตัดเกรด D! เพราะ 95 ≥ 50 เป็นจริง โปรแกรมเจอจริงตัวแรกก็หยุดเลย ไม่สนใจว่าข้างล่างมีเงื่อนไขที่ "จริงกว่า" รออยู่ — กฎ: เรียงจากเงื่อนไขที่แคบ/สูงสุดลงไปกว้างสุดเสมอ

The switch Statement

🎛️ switch–case — เมนูทางเลือกฉบับภาษาซี

เมื่อทางเลือกเป็น "ค่าตายตัว" (เมนู 1 2 3, เกรด A B C) ภาษาซีมีเครื่องมือเฉพาะ: switch (ตัวแปร) กระโดดไป case ที่ค่าตรงทันที — แต่มีกติกาเหล็ก: จบทุก case ด้วย break; ไม่งั้นไหลทะลุลงไปทำ case ถัดไปด้วย!

ตัวอย่างที่ 6.3 — menu.c เครื่องดื่มอัตโนมัติ (ผู้ใช้กด 2)
int menu;
scanf("%d", &menu);
switch (menu) {
    case 1:
        printf("กาแฟร้อน 35 บาท\n");
        break;
    case 2:
        printf("ชาเย็น 30 บาท\n");
        break;
    case 3:
        printf("โกโก้ 40 บาท\n");
        break;
    default:
        printf("ไม่มีเมนูนี้\n");
}

🆚 switch vs else if — เลือกอันไหน?

switch: ค่าตายตัวเป๊ะ ๆ (เมนู 1 2 3, ตัวอักษร 'A' 'B') — อ่านง่าย เร็ว
else if: เงื่อนไขเป็น "ช่วง" (score >= 80) หรือซับซ้อน — switch ทำไม่ได้

⚠️ switch ใช้ได้กับ int และ char เท่านั้น — ใช้กับ float หรือช่วงค่าไม่ได้
💡 default คือ else ของ switch: รับทุกค่าที่ไม่ตรง case ไหนเลย

🕳️

ลืม break = ไหลทะลุ (fall-through)

ถ้า case 2 ไม่มี break; โปรแกรมจะพิมพ์ "ชาเย็น" แล้ว ไหลลงไปพิมพ์ "โกโก้" ต่อทันที โดยไม่ตรวจค่าอะไรอีก — ข้อสอบชอบหลอกจุดนี้ที่สุดในเรื่อง switch

Nested Conditions

🪆 เงื่อนไขซ้อนเงื่อนไข (Nested if)

ใส่ if ไว้ "ข้างใน" { } ของ if อีกชั้นได้ — ใช้เมื่อการตัดสินใจมีลำดับขั้น เช่น ตรวจสิทธิ์เช่ารถ: ต้องอายุถึง "ก่อน" แล้วค่อยดูใบขับขี่

ตัวอย่างที่ 6.4 — rental.c อายุ 20 / มีใบขับขี่ (license = 1)
int age = 20;
int has_license = 1;   // 1 = มี, 0 = ไม่มี
if (age >= 18) {
    if (has_license) {
        printf("เช่ารถได้\n");
    } else {
        printf("อายุถึง แต่ต้องมีใบขับขี่\n");
    }
} else {
    printf("อายุไม่ถึง 18 ปี\n");
}

🆚 ซ้อน vs && — เลือกอันไหน?

ถ้าผลลัพธ์ "ผ่าน/ไม่ผ่าน" เหมือนกันหมด ใช้ && สั้นกว่า:
if (age >= 18 && has_license)

แต่ถ้าต้องการ ข้อความปฏิเสธต่างกัน ในแต่ละกรณี (อายุไม่ถึง vs ไม่มีใบขับขี่) ต้องซ้อนเพื่อแยกแขนงให้ครบ — กฎง่าย ๆ: ผลเหมือนใช้ && ผลต่างใช้ซ้อน

💡 สังเกต if (has_license) — ไม่ต้องเขียน == 1 เพราะ 1 ก็คือจริงอยู่แล้ว
⚠️ ซ้อนเกิน 3 ชั้นเริ่มอ่านยาก ควรคิดโครงสร้างใหม่

Putting It Together

⚡ ใช้งานจริง: ค่าไฟฟ้าแบบขั้นบันได

จำงานวาดผังงานค่าไฟในคาบที่ 3 ได้ไหม? วันนี้อัปเกรดเป็น 3 ขั้นบันไดด้วย else if — ยิ่งใช้มาก หน่วยที่เกินมายิ่งแพง ลองเลื่อนจำนวนหน่วยแล้วดูว่าโปรแกรมเข้าเงื่อนไขไหน

ตัวอย่างที่ 6.5 — electric_bill.c
int units;
float bill;
scanf("%d", &units);
if (units <= 150) {
    bill = units * 3.25;
} else if (units <= 400) {
    bill = 150 * 3.25 + (units - 150) * 4.22;
} else {
    bill = 150 * 3.25 + 250 * 4.22 + (units - 400) * 4.42;
}
printf("ค่าไฟ %.2f บาท\n", bill);

Check Your Understanding

📝 แบบทดสอบท้ายคาบ (10 ข้อ)

จำหลักไว้: ไล่บนลงล่าง เจอจริงแล้วหยุด · ; ท้าย if คือหายนะ · switch ต้องมี break

📘 อ่านก่อนสอบ · สรุปประเด็นสำคัญ

คาบนี้เน้น "ลำดับเงื่อนไข" และกับดักไวยากรณ์ของ if/switch — ระวัง 2 กับดักคลาสสิก: ; หลัง if และลืม break

1

บล็อก { } กำหนดขอบเขตของ if

คำสั่งที่อยู่ใน { } เท่านั้นที่ถือเป็นบล็อกของ if ถ้าไม่ใส่ปีกกา จะคุมแค่บรรทัดเดียว

2

กับดัก: ใส่ ; หลังเงื่อนไข if

if (speed > 90); — เซมิโคลอนทำให้ if มี "บล็อกว่าง" บล็อกถัดมาจึงทำงานทุกครั้งไม่ว่าเงื่อนไขจริงหรือเท็จ

if (speed > 90);     // ผิด! มี ; เกิน
{ printf("ใบสั่ง"); }  // พิมพ์เสมอ

3

if–else if–else ทำงานบล็อกเดียว

ทั้งชุดจะมีบล็อกถูกทำงาน เพียง 1 บล็อก · ลำดับเงื่อนไขสำคัญมาก เช่นตัดเกรด ถ้าเอา D (>=50) ขึ้นก่อน คน 95 คะแนนจะได้ D เพราะเข้าเงื่อนไขแรกที่จริง

if (score >= 80) ...  // A — ต้องเช็คมากไปน้อย
else if (score >= 70) ... // 68 → ตกมาได้ C

4

switch–case และ break

เหมาะกับการเทียบค่า "เท่ากับ" หลายค่าแบบเจาะจง (เช่นเมนู 1/2/3) · ลืม break จะ "ไหลลง" ทำ case ถัดไปต่อ (fall-through)

switch (menu) {
  case 1: ...; break;   // ขาด break = ไหลลง case 2
  default: ...;
}

5

เงื่อนไขซ้อนด้วย &&

รวมหลายเงื่อนไขที่ต้องจริงพร้อมกัน เช่น age >= 18 && has_license == 1

💡

จุดที่ข้อสอบชอบถาม

สองกับดักที่ออกบ่อย: ; หลัง if(...) และ ลืม break ใน switch — กับโจทย์ "สลับลำดับเงื่อนไขแล้วผลเปลี่ยน"
